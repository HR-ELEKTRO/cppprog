%>myTitlepage
% \myTitlepage definieert de titlepage
% 1 = extra ruimte vóór titel ebook versie
% 2 = extra ruimte vóór titel papieren versie
% 3 = fontsize titel ebook versie
% 4 = fontsize titel papieren versie
% 5 = subtitle (wordt na titel geplaatst)
% 6 = optional argument \includegraphics ebook versie
% 7 = optional argument \includegraphics papieren versie
% 8 = bestandsnaam plaatje (kan ook leeg gelaten worden) 
\newcommand{\myTitlepage}[8]{
    \newlength{\titlefontlinespace}
    \newlength{\titlefontlinespaceebook}
    \setlength{\titlefontlinespace}{#4}
    \setlength{\titlefontlinespaceebook}{#3}
    \addtolength{\titlefontlinespace}{.167\titlefontlinespace}
    \addtolength{\titlefontlinespaceebook}{.167\titlefontlinespaceebook}
    \begin{titlepage}
        \centering
        \iftoggle{ebook} {
            \vspace*{#1}
            {\sectionfont\fontsize{#3}{\titlefontlinespaceebook}\selectfont
                \title{}\par
            }
        }{
            \vspace*{#2}
            {\sectionfont\fontsize{#4}{\titlefontlinespace}\selectfont
                \title{}\par
            }
        }
        \par
        #5
        \ifstrempty{#8}{}{
            \vspace{\stretch{1}}
            \iftoggle{ebook} {
                \includegraphics[#6]{figs/#8}
            }{
                \includegraphics[#7]{figs/#8}
            }
        }
        \par
        \vspace{\stretch{1}}
        \begin{minipage}[c]{.5\linewidth}
            \centering
            \iftoggle{ebook} {
                \includegraphics[width=.8\linewidth]{figs/HHS_grijs_groen_fc}
            }{
                \includegraphics[width=\linewidth]{figs/HHS_grijs_groen_fc}
            }
        \end{minipage}\hfill
        \begin{minipage}[c]{.5\linewidth}
            \begin{spacing}{1.3}
                \raggedleft
                \author\\
                De Haagse Hogeschool\\
                Opleiding \opleiding\\
                \today\\
    	        \mail
            \end{spacing}
        \end{minipage}
    \end{titlepage}
    \iftoggle{ebook}{
        \setcounter{page}{2}
    }{
    }
}

%>myFigure
% iets meer ruimte boven en minder onder figuur:
\newenvironment{myFigure}[1][!htbp]{
    \begin{figure}[#1]
    \vspace{.2\baselineskip}%
    \def\endfigspace{
        \ifstrequal{#1}{H}{
            \vspace{-.8\baselineskip}%
        }{
            \vspace{-.2\baselineskip}%
        }
    }
}{
	\endfigspace{}%
    \end{figure}
}
%<myFigure

% iets meer ruimte boven en onder tabel:

    \vspace{.2\baselineskip}%
    \def\endtablespace{%
        \ifstrequal{#1}{H}{
            \vspace{-.4\baselineskip}%
        }{
            \vspace{.2\baselineskip}%
        }
    }
}{
	\endtablespace{}%
    \end{table}
}
%<myTable

%>figuur
% commando voor figuur
% figuren staan in subdir figs
% #1 OPTIONAL = placement argument voor figure b.v. tb or H 
% #2 = width of scale optie voor \includegraphics
% #3 = filenaam (zonder extensie); labelnaam=fig:filenaam
% #4 = caption tekst
\newcommand{\figuur}[4][!htbp]{
    \begin{myFigure}[#1]
        \centering%
        \includegraphics[#2]{figs/#3}
        \caption{#4}
        \label{fig:#3}
    \end{myFigure}
}
%<figuur

%>tabel
% commando voor tabel
% #1 OPTIONAL = placement argument for figure e.g. tb or H 
% #2 = caption tekst
% #3 = labelnaam=tab:#3
% #4 = tabu kolom definities
% #5 = kop tabel (eerste regel)
% #6 = inhoud tabel (rest van de regels)
\newcommand{\tabel}[6][!htbp]{
    \begin{myTable}[#1]
        \centering%
        \caption{#2}
        \label{tab:#3}
        \begin{tabu}{#4}
            \hline
            \rowfont[c]{\rmfamily\bfseries} #5\\ \hline
            #6\\ \hline
        \end{tabu}
    \end{myTable}
}

% variant voor brede tabel
% tabel wordt geschaald zodat hij past
\newcommand{\tabelmaxpaginabreedgeschaald}[6][!htbp]{
    \begin{myTable}[#1]
        \centering%
        \caption{#2}
        \label{tab:#3}
	    \begin{adjustbox}{max width=\textwidth}%
	        \begin{tabu}{#4}
	            \hline
	            \rowfont[c]{\bfseries} #5\\ \hline
	            #6\\ \hline
	        \end{tabu}%
	    \end{adjustbox}
    \end{myTable}
}

% variant voor lange tabel
% #1 = OPTIONAL position argument for longtabu c, l or r 
% #2 = caption tekst
% #3 = labelnaam=tab:#3
% #4 = tabu kolom definities
% #5 = kop tabel (eerste regel)
% #6 = inhoud tabel (rest van de regels)
% #7 = aantal kolommen
\newcommand{\langetabel}[7][c]{
    \begin{center}
        \vspace*{.2\baselineskip}%
        \begin{longtabu}[#1]{#4}
            \caption{#2}\label{tab:#3} \\
            \hline
            \rowfont[c]{\rmfamily\bfseries} #5\\ \hline
            \endfirsthead
            \caption[]{Vervolg. #2}\\
            \hline
            \rowfont[c]{\rmfamily\bfseries} #5\\ \hline
            \endhead
            \multicolumn{#7}{r}{\sffamily\small Zie vervolg op volgende pagina.}\\
            \endfoot
            \hline
            \endlastfoot
            #6 
        \end{longtabu}
        \vspace{.2\baselineskip}%
    \end{center}
}
